## UAT: Full PKCE Flow\n\n```python\nimport httpx, secrets, hashlib, base64\n\n# 1. Register client\nhttpx.post('http://localhost:8001/register', json={\n    'client_id': 'my-client', 'client_secret': 'my-secret',\n    'redirect_uri': 'http://localhost/cb', 'scope': 'tools:call'\n})\n\n# 2. Generate PKCE pair\nverifier = secrets.token_urlsafe(32)\nchallenge = base64.urlsafe_b64encode(hashlib.sha256(verifier.encode()).digest()).rstrip(b'=').decode()\n\n# 3. Authorize\nresp = httpx.get('http://localhost:8001/authorize', params={\n    'client_id': 'my-client', 'code_challenge': challenge,\n    'code_challenge_method': 'S256', 'scope': 'tools:call', 'user_id': 'alice'\n})\ncode = resp.json()['code']\n\n# 4. Exchange\nresp = httpx.post('http://localhost:8001/token', json={\n    'code': code, 'code_verifier': verifier,\n    'client_id': 'my-client', 'client_secret': 'my-secret',\n    'grant_type': 'authorization_code'\n})\ntoken = resp.json()['access_token']\n\n# 5. Call protected MCP endpoint\nresp = httpx.post('http://localhost:8000/mcp',\n    json={'jsonrpc':'2.0','id':1,'method':'tools/list','params':{}},\n    headers={'Authorization': f'Bearer {token}'}\n)\nprint(resp.json())  # Expected: result.tools\n\n# 6. Confused deputy: use token with wrong audience\n# Expect: 401 invalid audience\n```
